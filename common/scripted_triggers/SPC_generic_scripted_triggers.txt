can_core_nation_through_compliance = {	#可核心化占领
	any_of_scopes = {
		array = potential_cores
		original_tag = FROM
	}
	FROM = {
		NOT = { has_country_flag = can_core_nation_through_compliance_tr }	#纯粹tt
		hidden_trigger = {
			any_core_state = {
				custom_trigger_tooltip = {
					if = {
						limit = { NOT = { is_core_of = ROOT } }
						NOT = {
							is_in_array = {
								array = ROOT.unpotential_cores
								value = THIS.id
							}
						}
					}
					else = {
						always = no
					}
				}
			}
		}
	}
}

can_core_nation_through_compliance_state_ROOT = {
	NOT = { is_core_of = ROOT }
	any_of_scopes = {
		array = core_countries
		is_in_array = {
			ROOT.potential_cores = original_tag
		}
	}
}

cannot_core_state_through_compliance_state_ROOT = {
	NOT = {
		is_in_array = {
			array = ROOT.unpotential_cores
			value = THIS.id
		}
	}
}
