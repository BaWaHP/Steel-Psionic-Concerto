LNR_grey_support_up = {
	custom_effect_tooltip = LNR_grey_support_up_tt
	if = {
		limit = { has_idea = LNR_grey_support_5 }
		swap_ideas = {
			remove_idea = LNR_grey_support_5
			add_idea = LNR_grey_support_4
		}
	}
	else_if = {
		limit = { has_idea = LNR_grey_support_4 }
		swap_ideas = {
			remove_idea = LNR_grey_support_4
			add_idea = LNR_grey_support_3
		}
	}
	else_if = {
		limit = { has_idea = LNR_grey_support_3 }
		swap_ideas = {
			remove_idea = LNR_grey_support_3
			add_idea = LNR_grey_support_2
		}
	}
	else_if = {
		limit = { has_idea = LNR_grey_support_2 }
		swap_ideas = {
			remove_idea = LNR_grey_support_2
			add_idea = LNR_grey_support_1
		}
	}
}
