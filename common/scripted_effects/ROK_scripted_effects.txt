#ROK选举事件结束效果 确定哪个政党获胜
SPC_ROK_vote_event_end_effect = {
	if = {
		limit = {
			AND = {
				check_variable = { rok_worldunion_democrat_votes < 3 }
				check_variable = { rok_liberal_republican_votes < 3 }
				check_variable = { rok_conservative_republican_votes < 3 }
			}
		}
		country_event = { id = ROK.5 days = 7 } #选举陷入僵局
	}
	
	if = {
		limit = {
			check_variable = { rok_worldunion_democrat_votes > 2 }
		}
		country_event = { id = ROK.13 days = 7 } #选举 社民党获胜 
	}
	
	if = {
		limit = {
			check_variable = { rok_liberal_republican_votes > 2 }
		}
		country_event = { id = ROK.14 days = 7 } #选举 新民主党获胜 
	}
	
	if = {
		limit = {
			check_variable = { rok_conservative_republican_votes > 2 }
		}
		country_event = { id = ROK.15 days = 7 } #选举 民主党获胜 
	}
}

ROK_temp_pass_check_effet = {
	set_temp_variable = { ROK_temp_pass_check = 0 }
	if = {
		limit = { check_variable = { ROK_faction_alt_1_var = 1 } }
		add_to_temp_variable = { ROK_temp_pass_check = -1 }
	}
	else_if = {
		limit = { check_variable = { ROK_faction_alt_1_var = 3 } }
		add_to_temp_variable = { ROK_temp_pass_check = 1 }
	}
	if = {
		limit = { check_variable = { ROK_faction_alt_2_var = 1 } }
		add_to_temp_variable = { ROK_temp_pass_check = -1 }
	}
	else_if = {
		limit = { check_variable = { ROK_faction_alt_2_var = 3 } }
		add_to_temp_variable = { ROK_temp_pass_check = 1 }
	}
	if = {
		limit = { check_variable = { ROK_faction_alt_3_var = 1 } }
		add_to_temp_variable = { ROK_temp_pass_check = -1 }
	}
	else_if = {
		limit = { check_variable = { ROK_faction_alt_3_var = 3 } }
		add_to_temp_variable = { ROK_temp_pass_check = 1 }
	}
	if = {
		limit = { check_variable = { ROK_faction_alt_4_var = 1 } }
		add_to_temp_variable = { ROK_temp_pass_check = -1 }
	}
	else_if = {
		limit = { check_variable = { ROK_faction_alt_4_var = 3 } }
		add_to_temp_variable = { ROK_temp_pass_check = 1 }
	}
	if = {
		limit = { check_variable = { ROK_temp_pass_check > 0 } }
		custom_effect_tooltip = ROK_proposal_pass_tt
		ROK_proposal_is_being_voted_on_effect = yes
	}
	else = {
		if = {
			limit = { is_ai = yes }
			random_list = {
				80 = {
					ROK_proposal_is_being_voted_on_effect = yes
				}
			}
		}
		custom_effect_tooltip = ROK_proposal_fail_tt
	}
}
