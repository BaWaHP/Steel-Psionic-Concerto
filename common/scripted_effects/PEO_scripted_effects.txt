PEO_maritime_trade_administration_var_clamp = {
	clamp_variable = {
		var = PEO_flourished_ocean_trade_var
		min = 0
		max = 250
	}
}

PEO_maritime_trade_administration_remove_all = {
	if = {
		limit = { has_idea = PEO_flourished_ocean_trade_1 }
		remove_ideas = PEO_flourished_ocean_trade_1
	}
	if = {
		limit = { has_idea = PEO_flourished_ocean_trade_2 }
		remove_ideas = PEO_flourished_ocean_trade_2
	}
	if = {
		limit = { has_idea = PEO_flourished_ocean_trade_3 }
		remove_ideas = PEO_flourished_ocean_trade_3
	}
	if = {
		limit = { has_idea = PEO_flourished_ocean_trade_4 }
		remove_ideas = PEO_flourished_ocean_trade_4
	}
	if = {
		limit = { has_idea = PEO_flourished_ocean_trade_5 }
		remove_ideas = PEO_flourished_ocean_trade_5
	}
}

PEO_maritime_trade_administration_idea_change = {
	if = {
		limit = {
			check_variable = { PEO_flourished_ocean_trade_var < 50 }
		}
		add_ideas = PEO_flourished_ocean_trade_1
	}
	else_if = {
		limit = {
			check_variable = { PEO_flourished_ocean_trade_var > 49 }
			check_variable = { PEO_flourished_ocean_trade_var < 100 }
		}
		add_ideas = PEO_flourished_ocean_trade_2
	}
	else_if = {
		limit = {
			check_variable = { PEO_flourished_ocean_trade_var > 99 }
			check_variable = { PEO_flourished_ocean_trade_var < 150 }
		}
		add_ideas = PEO_flourished_ocean_trade_3
	}
	else_if = {
		limit = {
			check_variable = { PEO_flourished_ocean_trade_var > 149 }
			check_variable = { PEO_flourished_ocean_trade_var < 200 }
		}
		add_ideas = PEO_flourished_ocean_trade_4
	}
	else_if = {
		limit = {
			check_variable = { PEO_flourished_ocean_trade_var > 199 }
			check_variable = { PEO_flourished_ocean_trade_var < 251 }
		}
		add_ideas = PEO_flourished_ocean_trade_5
	}
}

PEO_current_congress_icon_frame_set = {
	set_temp_variable = { congress_temp = PEO_congress_seats_supporter }
	divide_temp_variable = { congress_temp = PEO_congress_seats_total }
	multiply_temp_variable = { congress_temp = 21 }
	round_temp_variable = congress_temp
	if = {
		limit = { check_variable = { congress_temp > 0 } }
		clamp_temp_variable = {
			var = congress_temp
			min = 1
			max = 21
		}
	}
	else = { set_temp_variable = { congress_temp = 1 } }
	set_variable = { PEO_current_congress_icon_frame = congress_temp }
}
