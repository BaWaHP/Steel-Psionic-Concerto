generic_territory_puppet_effect_special = {
	if = {
		limit = { ROOT = { tag = SFR has_government = revolutionary_communism } }
		create_country_leader = {
			name = SFR_puppet_leader_mil_gov
			picture = "GFX_SFR_puppet_leader_mil_gov"
			ideology = revolutionary_communism_subtype
			traits = { SFR_military_control }
		}
		load_focus_tree = generic_focus
		SFR = { add_to_faction = PREV }
	}
}

generic_territory_return_and_coring_effect = {
	every_owned_state = {
		if = {
			limit = { OR = { is_core_of = ROOT is_claimed_by = ROOT } }
			transfer_state_to = ROOT
		}
		if = {
			limit = { NOT = { is_core_of = PREV } }
			add_core_of = PREV
		}
	}
}

generic_territory_add_popularty_effect = {
	add_popularity = {
		ideology = var:current_party_ideology_group
		popularity = 0.8
	}
}

generic_territory_puppet_effect = {
	set_autonomy = {
		target = event_target:released_puppet_country
		autonomy_state = autonomy_puppet
	}
	event_target:released_puppet_country = {
		#load_focus_tree = generic_focus
		generic_territory_return_and_coring_effect = yes
		generic_territory_add_popularty_effect = yes
		set_country_flag = generic_territory_has_been_rebuilt
		generic_territory_puppet_effect_special = yes
	}
}
