WUN_yarlig_check = {
	clamp_variable = {
		var = WUN_yarlig_var
		min = 0
		max = 350000
	}
	hidden_effect = {
		if = {
			limit = { check_variable = { WUN_yarlig_var = 350000 } }
			WUN_random_buff = yes
		}
	}
}

WUN_random_buff = {
	set_variable = { WUN_yarlig_var = 0 }
	set_temp_variable = { WUN_get_current_buff_count = WUN_buff_count }
	if = {
		limit = { check_variable = { WUN_buff_count = 5 } }
		add_political_power = 1
		#country_event = WUN.10
	}
	else = {
		random_list = {
			20 = {
				if = {
					limit = { NOT = { has_idea = WUN_sarcoma } }
					add_ideas = WUN_sarcoma
					country_event = WUN.100
				}
			}
			20 = {
				if = {
					limit = { NOT = { has_idea = WUN_callus } }
					add_ideas = WUN_callus
					country_event = WUN.100
				}
			}
			20 = {
				if = {
					limit = { NOT = { has_idea = WUN_tissue_fluid } }
					add_ideas = WUN_tissue_fluid
					country_event = WUN.100
				}
			}
			20 = {
				if = {
					limit = { NOT = { has_idea = WUN_granulation } }
					add_ideas = WUN_granulation
					country_event = WUN.100
				}
			}
			20 = {
				if = {
					limit = { NOT = { has_idea = WUN_mound_of_flesh } }
					add_ideas = WUN_mound_of_flesh
					country_event = WUN.100
				}
			}
		}
		if = {
			limit = { check_variable = { WUN_get_current_buff_count = WUN_buff_count } }
			WUN_random_buff = yes
		}
	}
}
