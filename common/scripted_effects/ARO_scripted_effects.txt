ARO_mfL_status_update = {
	ROOT = {
		every_controlled_state = {
			limit = {
				OR = {
					state = 72
					state = 73
					state = 74
					state = 75
					state = 79
					state = 80
				}
			}
			if = {
				limit = {
					free_building_slots = {
						building = arms_factory
						size > 0
						include_locked = no
					}
				}
				set_variable = { ARO_mfL_state_status = 1 }
			}
			else = {
				set_variable = { ARO_mfL_state_status = 0 }
			}
		}
		every_controlled_state = {
			limit = {
				OR = {
					state = 81
					state = 82
					state = 83
					state = 84
					state = 86
				}
			}
			set_variable = { ARO_mfL_state_status = 0 }
			if = {
				limit = {
					arms_factory > 0
				}
				add_to_variable = { ARO_mfL_state_status = 1 }
			}
			if = {
				limit = {
					industrial_complex > 0
				}
				add_to_variable = { ARO_mfL_state_status = 1 }
			}
			if = {
				limit = {
					infrastructure > 0
				}
				add_to_variable = { ARO_mfL_state_status = 1 }
			}
		}
	}
