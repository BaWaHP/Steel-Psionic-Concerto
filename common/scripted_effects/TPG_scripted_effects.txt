#公国工业发展度增加
TPG_industrial_development_degree_up = {
	if = {
	    limit = {
		    tag = TPG
			has_idea = TPG_industrial_development_degree_1_i
		}
		swap_ideas = {
			remove_idea = TPG_industrial_development_degree_1_i
			add_idea = TPG_industrial_development_degree_2_i
		}
	}
	else_if = {
	    limit = {
		    tag = TPG
			has_idea = TPG_industrial_development_degree_2_i
		}
		swap_ideas = {
			remove_idea = TPG_industrial_development_degree_2_i
			add_idea = TPG_industrial_development_degree_3_i
		}
	}
	else_if = {
	    limit = {
		    tag = TPG
			has_idea = TPG_industrial_development_degree_3_i
		}
		swap_ideas = {
			remove_idea = TPG_industrial_development_degree_3_i
			add_idea = TPG_industrial_development_degree_4_i
		}
	}
	else_if = {
	    limit = {
		    tag = TPG
			has_idea = TPG_industrial_development_degree_4_i
		}
		swap_ideas = {
			remove_idea = TPG_industrial_development_degree_4_i
			add_idea = TPG_industrial_development_degree_5_i
		}
	}
	else = {
	    
	}
	custom_effect_tooltip = TPG_industrial_development_degree_up_tt
}
