RLN_house_of_cards_refresh_vars = {
	RLN_house_of_cards_refresh_leadership = yes

	set_variable = { RLN_house_of_cards_largest_faction_count = 0 }
	if = {
		limit = { has_country_flag = RLN_largest_faction_1 }
		add_to_variable = { RLN_house_of_cards_largest_faction_count = 1 }
		if = {
			limit = { has_government = pioneer_communism }
			set_country_flag = RLN_largest_faction_controls
		}
	}
	if = {
		limit = { has_country_flag = RLN_largest_faction_2 }
		add_to_variable = { RLN_house_of_cards_largest_faction_count = 1 }
		if = {
			limit = { has_government = revolutionary_communism }
			set_country_flag = RLN_largest_faction_controls
		}
	}
	if = {
		limit = { has_country_flag = RLN_largest_faction_3 }
		add_to_variable = { RLN_house_of_cards_largest_faction_count = 1 }
		if = {
			limit = { has_government = worldunion_democrat }
			set_country_flag = RLN_largest_faction_controls
		}
	}
	if = {
		limit = { has_country_flag = RLN_largest_faction_4 }
		add_to_variable = { RLN_house_of_cards_largest_faction_count = 1 }
		if = {
			limit = { has_government = national_socialist }
			set_country_flag = RLN_largest_faction_controls
		}
	}

	if = {
		limit = { check_variable = { RLN_house_of_cards_largest_faction_count = 1 } }
		if = {
			limit = { has_country_flag = RLN_largest_faction_controls }
			clr_country_flag = RLN_largest_faction_controls
			set_variable = { RLN_one_party_four_factions_ppf = 0.1 }
			set_variable = { RLN_one_party_four_factions_sf = 0 }
		}
		else = {
			set_variable = { RLN_one_party_four_factions_ppf = -0.1 }
			set_variable = { RLN_one_party_four_factions_sf = 0 }
		}
	}
	else = {
		set_variable = { RLN_one_party_four_factions_ppf = -0.1 }
		set_variable = { RLN_one_party_four_factions_sf = -0.1 }
	}

	force_update_dynamic_modifier = yes
}

RLN_election_result_effect = {		#赤色特尼安主席选举效果
	if = {
		limit = {
			check_variable = {
				var = RLN_principlists_influence
				value = RLN_financists_influence
				compare = greater_than_or_equals
			}
			check_variable = {
				var = RLN_principlists_influence
				value = RLN_militarists_influence
				compare = greater_than_or_equals
			}
		}
		RLN_principlists_election_win_effect = yes
	}
	else_if = {
		limit = {
			check_variable = {
				var = RLN_financists_influence
				value = RLN_principlists_influence
				compare = greater_than_or_equals
			}
			check_variable = {
				var = RLN_financists_influence
				value = RLN_militarists_influence
				compare = greater_than_or_equals
			}
		}
		RLN_financists_election_win_effect = yes
	}
	else_if = {
		limit = {
			check_variable = {
				var = RLN_militarists_influence
				value = RLN_principlists_influence
				compare = greater_than_or_equals
			}
			check_variable = {
				var = RLN_militarists_influence
				value = RLN_financists_influence
				compare = greater_than_or_equals
			}
		}
		RLN_militarists_election_win_effect = yes
	}
}
