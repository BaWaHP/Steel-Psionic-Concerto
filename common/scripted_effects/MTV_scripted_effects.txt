MTV_civil_idea_worse = {
	custom_effect_tooltip = MTV_civil_idea_worse_tt
	MTV_civil_idea_worse_update = yes
}

MTV_civil_idea_worse_update = {
	if = {
		limit = { has_idea = MTV_man_in_hell_1 }
		swap_ideas = {
			remove_idea = xMTV_man_in_hell_1
			add_idea = MTV_man_in_hell_2
		}
	}
	else_if = {
		limit = { has_idea = MTV_man_in_hell_2 }
		swap_ideas = {
			remove_idea = MTV_man_in_hell_2
			add_idea = MTV_man_in_hell_3
		}
	}
	else = {
		add_stability = -0.1
		add_war_support = -0.1
	}
}
