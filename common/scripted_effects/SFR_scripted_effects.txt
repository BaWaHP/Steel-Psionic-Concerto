SPC_SFR_startup_effect = {
	if = {
		limit = { tag = SFR }
		###介绍页面###
		set_variable = { SPC_intro_screen = 1 }
		###普及度修正部分###
		set_variable = { magic_development_monthly_buff = 0 }
		set_variable = { science_development_monthly_buff = 0 }
		every_owned_state = {
			set_variable = { state_magic_monthly_buff = 0 }
			set_variable = { state_science_monthly_buff = 0 }
		}
		###政府法案###
		add_ideas = service_by_requirement
		add_ideas = no_standard_at_all
		add_ideas = partial_economic_mobilisation
		add_ideas = closed_economy
		add_ideas = high_taxes
		add_ideas = NP_total_nationalization
		add_ideas = defensive_tatic
		###发展度###
		add_ideas = handcraft_industry
		add_ideas = self_efficient_agriculture
		add_ideas = unorganized_army
		add_ideas = normally_living
		add_ideas = uncomfortable
		add_ideas = fake_exists
		###工业政策###
		add_ideas = EP_barely_nothing
		add_ideas = RP_asisting_research
		add_ideas = IP_nationalization
		add_ideas = WP_better_laws
		add_ideas = AP_basic_funds
		add_ideas = RP_no_funds
		###政府政策###
		add_ideas = MP_funding_media
		add_ideas = SP_sybolized_funds
		add_ideas = MP_research_only
		add_ideas = SPP_no_secreat_police
		add_ideas = FP_joining_the_army
		add_ideas = EP_sharing_textbooks
		###种族和特殊魔法###
		add_ideas = SM_no_special_magic
		add_ideas = LR_humans
		add_ideas = SR_LingRen
		add_ideas = RAP_lower_suppression
		add_ideas = RLP_secularism
		add_ideas = SXP_traditional_policy
		###发展度设置###
		SPC_development_update_effect = yes
		###其它开局设置###
		if = {
			limit = { has_template = "皇家卫队" }
			delete_unit_template_and_units = { division_template = "皇家卫队" }
		}
		if = {
			limit = { has_template = "远乡正规军" }
			delete_unit_template_and_units = { division_template = "远乡正规军" }
		}
		if = {
			limit = { has_template = "远乡边防军" }
			delete_unit_template_and_units = { division_template = "远乡边防军" }
		}
		if = {
			limit = { has_template = "远乡装甲部队" }
			delete_unit_template_and_units = { division_template = "远乡装甲部队" }
		}
		load_oob = "SFR"
		activate_mission = SFH_taking_actions
		activate_mission = SFR_taking_actions
		activate_mission = QTA_taking_actions
		SFR_maintain_the_front_setup = yes
		set_country_flag = SFR_opposition_bop_on
		set_variable = { SFR_left_right_influence = 0 } #正是左，右是负
	}
}

SFR_add_front_unity_ideas = {
	if = {
		limit = {
			check_variable = { SFR_maintain_the_front_unity > 79 }
			check_variable = { SFR_maintain_the_front_unity < 101 }
		}
		add_ideas = SFR_front_unity_idea_1
	}
	if = {
		limit = {
			check_variable = { SFR_maintain_the_front_unity > 59 }
			check_variable = { SFR_maintain_the_front_unity < 80 }
		}
		add_ideas = SFR_front_unity_idea_2
	}
	if = {
		limit = {
			check_variable = { SFR_maintain_the_front_unity > 39 }
			check_variable = { SFR_maintain_the_front_unity < 60 }
		}
		add_ideas = SFR_front_unity_idea_3
	}
	if = {
		limit = {
			check_variable = { SFR_maintain_the_front_unity > 19 }
			check_variable = { SFR_maintain_the_front_unity < 40 }
		}
		add_ideas = SFR_front_unity_idea_4
	}
	if = {
		limit = {
			check_variable = { SFR_maintain_the_front_unity > -1 }
			check_variable = { SFR_maintain_the_front_unity < 20 }
		}
		add_ideas = SFR_front_unity_idea_5
	}
}
