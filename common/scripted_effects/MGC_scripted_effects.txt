MGC_spring_conspiracy_influence_update = {
	every_character = {
		limit = { has_variable = MGC_spring_conspiracy_supporter }
		if = {
			limit = { check_variable = { MGC_spring_conspiracy_supporter = 1 } }
			add_to_variable = { MGC.MGC_spring_conspiracy_influence_2 = 5 }
		}
		else_if = {
			limit = { check_variable = { MGC_spring_conspiracy_supporter = 2 } }
			add_to_variable = { MGC.MGC_spring_conspiracy_influence_1 = 5 }
		}
		else_if = {
			limit = { check_variable = { MGC_spring_conspiracy_supporter = 3 } }
			add_to_variable = { MGC.MGC_spring_conspiracy_influence_3 = 5 }
		}
	}
}

MGC_spring_conspiracy_tension_update = {
	clamp_variable = {
		var = MGC_spring_conspiracy_tension
		min = 0
		max = 100
	}
	if = {
		limit = { check_variable = { MGC_spring_conspiracy_tension = 100 } }
		country_event = {
			id = MGC.0
			days = 1
		}
	}
}
