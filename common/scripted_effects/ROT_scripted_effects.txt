ROT_econ_depression_effect = {
	custom_effect_tooltip = ROT_econ_depression_tt
	if = {
		limit = { has_idea = ROT_econ_depression_1 }
		swap_ideas = {
			remove_idea = ROT_econ_depression_1
			add_idea = ROT_econ_depression_2
		}
	}
	else_if = {
		limit = { has_idea = ROT_econ_depression_2 }
		swap_ideas = {
			remove_idea = ROT_econ_depression_2
			add_idea = ROT_econ_depression_3
		}
	}
	else_if = {
		limit = { has_idea = ROT_econ_depression_3 }
		swap_ideas = {
			remove_idea = ROT_econ_depression_3
			add_idea = ROT_econ_depression_4
		}
	}
	else_if = {
		limit = { has_idea = ROT_econ_depression_4 }
		swap_ideas = {
			remove_idea = ROT_econ_depression_4
			add_idea = ROT_econ_depression_5
		}
	}
	else = {
		remove_ideas = ROT_econ_depression_5
	}
}

ROT_a_national_revolution_setup = {
	set_country_flag = ROT_class_and_nation_on
	unlock_decision_category_tooltip = ROT_class_and_nation

	add_dynamic_modifier = { modifier = ROT_a_national_revolution }

	set_variable = { ROT_a_national_revolution_1 = 70 } #左翼分子
	set_variable = { ROT_a_national_revolution_2 = 40 } #社会精英
	set_variable = { ROT_a_national_revolution_3 = 50 } #联邦裔公民
	set_variable = { ROT_a_national_revolution_4 = 60 } #提特兰人
	set_variable = { ROT_a_national_revolution_5 = 55 } #行政效率
	set_variable = { ROT_a_national_revolution_6 = 20 } #民族凝聚力

	ROT_a_national_revolution_update = yes
}

ROT_a_national_revolution_update = {
	ROT_a_national_revolution_clamp = yes

	set_variable = { ROT_a_national_revolution_ppc = 0 }
	set_variable = { ROT_a_national_revolution_sf = 0 }
	set_variable = { ROT_a_national_revolution_pfegf = 0 }
	set_variable = { ROT_a_national_revolution_ifrf = 0 }
	set_variable = { ROT_a_national_revolution_cgf = 0 }
	set_variable = { ROT_a_national_revolution_plorp = 0 }
	set_variable = { ROT_a_national_revolution_cf = 0 }
	set_variable = { ROT_a_national_revolution_icf = 0 }
	set_variable = { ROT_a_national_revolution_aof = 0 }
	set_variable = { ROT_a_national_revolution_c = 0 }
	set_variable = { ROT_a_national_revolution_ppf = 0 }
	set_variable = { ROT_a_national_revolution_psb = 0 }

	set_variable = { ROT_a_national_revolution_sf1 = 0 }
	set_variable = { ROT_a_national_revolution_sf2 = 0 }
	set_variable = { ROT_a_national_revolution_sf3 = 0 }
	set_variable = { ROT_a_national_revolution_sf4 = 0 }
	set_variable = { ROT_a_national_revolution_sf5 = 0 }

	#左翼分子
	if = {
		limit = { check_variable = { ROT_a_national_revolution_1 > 79 } }
		add_to_variable = { ROT_a_national_revolution_pfegf = 0.05 }
		add_to_variable = { ROT_a_national_revolution_ppc = -0.1 }
		add_to_variable = { ROT_a_national_revolution_ifrf = 0.3 }
	}
	else_if = {
		limit = { check_variable = { ROT_a_national_revolution_1 < 21 } }
		add_to_variable = { ROT_a_national_revolution_ppc = 0.05 }
		add_to_variable = { ROT_a_national_revolution_sf = -0.1 }
		set_variable = { ROT_a_national_revolution_sf1 = -0.1 }
	}

	#社会精英
	if = {
		limit = { check_variable = { ROT_a_national_revolution_2 > 79 } }
		add_to_variable = { ROT_a_national_revolution_cgf = -0.2 }
		add_to_variable = { ROT_a_national_revolution_plorp = -0.1 }
	}
	else_if = {
		limit = { check_variable = { ROT_a_national_revolution_2 < 21 } }
		add_to_variable = { ROT_a_national_revolution_cgf = 0.2 }
		add_to_variable = { ROT_a_national_revolution_sf = -0.1 }
		set_variable = { ROT_a_national_revolution_sf2 = -0.1 }
	}

	#联邦裔公民
	if = {
		limit = { check_variable = { ROT_a_national_revolution_3 > 79 } }
		add_to_variable = { ROT_a_national_revolution_cf = 0.05 }
		add_to_variable = { ROT_a_national_revolution_icf = 0.05 }
	}
	else_if = {
		limit = { check_variable = { ROT_a_national_revolution_3 < 21 } }
		add_to_variable = { ROT_a_national_revolution_cf = -0.05 }
		add_to_variable = { ROT_a_national_revolution_sf = -0.1 }
		set_variable = { ROT_a_national_revolution_sf3 = -0.1 }
	}

	#提特兰人
	if = {
		limit = { check_variable = { ROT_a_national_revolution_4 > 79 } }
		add_to_variable = { ROT_a_national_revolution_c = 0.005 }
		add_to_variable = { ROT_a_national_revolution_aof = 0.1 }
	}
	else_if = {
		limit = { check_variable = { ROT_a_national_revolution_4 < 21 } }
		add_to_variable = { ROT_a_national_revolution_aof = -0.15 }
		add_to_variable = { ROT_a_national_revolution_sf = -0.15 }
		set_variable = { ROT_a_national_revolution_sf4 = -0.15 }
	}

	#行政效率
	if = {
		limit = { check_variable = { ROT_a_national_revolution_5 > 79 } }
		add_to_variable = { ROT_a_national_revolution_ppf = 0.1 }
	}
	else_if = {
		limit = { check_variable = { ROT_a_national_revolution_5 < 21 } }
		add_to_variable = { ROT_a_national_revolution_ppf = -0.1 }
		add_to_variable = { ROT_a_national_revolution_sf = -0.1 }
		add_to_variable = { ROT_a_national_revolution_psb = -0.2 }
		set_variable = { ROT_a_national_revolution_sf5 = -0.1 }
	}
}
