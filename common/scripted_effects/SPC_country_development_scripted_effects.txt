SPC_country_development_upgrade = {
	add_to_variable = { industrial_development_amount = industrial_development_process }
	add_to_variable = { agricultural_development_amount = agricultural_development_process }
	add_to_variable = { military_development_amount = military_development_process }
	add_to_variable = { living_standard_development_amount = living_standard_development_process }
	if = {
		limit = { has_war = yes }
		add_to_variable = { living_standard_development_amount = -4 }
	}
	add_to_variable = { environment_development_amount = environment_development_process }
	add_to_variable = { research_development_amount = research_development_process }
	#add_to_variable = { race_mixing_status_amount = race_mixing_status_process }
	###工业###
	if = {
		limit = { check_variable = { industrial_development_amount > 199 } }
		if = { limit = { NOT = { has_idea = innovative_industry } } country_event = CD.1 }
		set_variable = { industrial_development_amount = 0 }
	}
	if = {
		limit = { check_variable = { industrial_development_amount < -99 } }
		if = { limit = { NOT = { has_idea = primitive_industry } } country_event = CD.2 }
		set_variable = { industrial_development_amount = 199 }
	}
	#农业
	if = {
		limit = { check_variable = { agricultural_development_amount > 199 } }
		if = { limit = { NOT = { has_idea = mechanized_agriculture } } country_event = CD.3 }
		set_variable = { agricultural_development_amount = 0 }
	}
	if = {
		limit = { check_variable = { agricultural_development_amount < -99 } }
		if = { limit = { NOT = { has_idea = primitive_agriculture } } country_event = CD.4 }
		set_variable = { agricultural_development_amount = 199 }
	}
	#军事
	if = {
		limit = { check_variable = { military_development_amount > 199 } }
		if = { limit = { NOT = { has_idea = innovative_army } } country_event = CD.5 }
		set_variable = { military_development_amount = 0 }
	}
	if = {
		limit = { check_variable = { military_development_amount < -99 } }
		if = { limit = { NOT = { has_idea = primitive_army } } country_event = CD.6 }
		set_variable = { military_development_amount = 199 }
	}
	#生活水平
	if = {
		limit = { check_variable = { living_standard_development_amount > 199 } }
		if = { limit = { NOT = { has_idea = best_living } } country_event = CD.7 }
		set_variable = { living_standard_development_amount = 0 }
	}
	if = {
		limit = { check_variable = { living_standard_development_amount < -99 } }
		if = { limit = { NOT = { has_idea = death_everywhere } } country_event = CD.8 }
		set_variable = { living_standard_development_amount = 199 }
	}
	#环境程度
	if = {
		limit = { check_variable = { environment_development_amount > 199 } }
		if = { limit = { NOT = { has_idea = real_natural_environment } } country_event = CD.9 }
		set_variable = { environment_development_amount = 0 }
	}
	if = {
		limit = { check_variable = { environment_development_amount < -99 } }
		if = { limit = { NOT = { has_idea = hazard_zone } } country_event = CD.10 }
		set_variable = { environment_development_amount = 199 }
	}
	#学术氛围
	if = {
		limit = { check_variable = { research_development_amount > 199 } }
		if = { limit = { NOT = { has_idea = willing_research } } country_event = CD.11 }
		set_variable = { research_development_amount = 0 }
	}
	if = {
		limit = { check_variable = { research_development_amount < -99 } }
		if = { limit = { NOT = { has_idea = fuck_the_science } } country_event = CD.12 }
		set_variable = { research_development_amount = 199 }
	}

	set_variable = { industrial_development_frame = industrial_development_amount }
	multiply_variable = { industrial_development_frame = 0.5 }
	set_variable = { agricultural_development_frame = agricultural_development_amount }
	multiply_variable = { agricultural_development_frame = 0.5 }
	set_variable = { military_development_frame = military_development_amount }
	multiply_variable = { military_development_frame = 0.5 }
	set_variable = { living_standard_development_frame = living_standard_development_amount }
	multiply_variable = { living_standard_development_frame = 0.5 }
	set_variable = { environment_development_frame = environment_development_amount }
	multiply_variable = { environment_development_frame = 0.5 }
	set_variable = { research_development_frame = research_development_amount }
	multiply_variable = { research_development_frame = 0.5 }
}

SPC_country_development_ideas_change_industry_up = {
	if = {
		limit = {
			has_idea = primitive_industry
		}
		swap_ideas = {
			remove_idea = primitive_industry
			add_idea = handcraft_industry
		}
	}
	else_if = {
		limit = {
			has_idea = handcraft_industry
		}
		swap_ideas = {
			remove_idea = handcraft_industry
			add_idea = early_factory_industry
		}
	}
	else_if = {
		limit = {
			has_idea = early_factory_industry
		}
		swap_ideas = {
			remove_idea = early_factory_industry
			add_idea = factory_industry
		}
	}
	else_if = {
		limit = {
			has_idea = factory_industry
		}
		swap_ideas = {
			remove_idea = factory_industry
			add_idea = grouped_factory_industry
		}
	}
	else_if = {
		limit = {
			has_idea = grouped_factory_industry
		}
		swap_ideas = {
			remove_idea = grouped_factory_industry
			add_idea = innovative_industry
		}
	}
	if = {
		limit = {
			OR = {
				check_variable = { industrial_development_mod > 1 }
				check_variable = { industrial_development_mod < -1 }
			}
		}
		divide_variable = { industrial_development_mod = 2 }
		round_variable = industrial_development_mod
	}
	else = {
		set_variable = { industrial_development_mod = 0 }
	}
}
