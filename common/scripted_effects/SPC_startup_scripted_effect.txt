SPC_development_mod_var_set = {
	if = {
		limit = { NOT = { has_variable = industrial_development_mod } }
		set_variable = { industrial_development_mod = 0 }
	}
	if = {
		limit = { NOT = { has_variable = agricultural_development_mod } }
		set_variable = { agricultural_development_mod = 0 }
	}
	if = {
		limit = { NOT = { has_variable = military_development_mod } }
		set_variable = { military_development_mod = 0 }
	}
	if = {
		limit = { NOT = { has_variable = living_standard_development_mod } }
		set_variable = { living_standard_development_mod = 0 }
	}
	if = {
		limit = { NOT = { has_variable = environment_development_mod } }
		set_variable = { environment_development_mod = 0 }
	}
	if = {
		limit = { NOT = { has_variable = research_development_mod } }
		set_variable = { research_development_mod = 0 }
	}
}

SPC_development_update_effect = {
	set_variable = { industrial_development_process = 0 }
	set_variable = { agricultural_development_process = 0 }
	set_variable = { military_development_process = 0 }
	set_variable = { living_standard_development_process = -2 }
	set_variable = { environment_development_process = 0 }
	set_variable = { research_development_process = 0 }
	SPC_development_mod_var_set = yes

	add_to_variable = { industrial_development_process = industrial_development_mod }
	add_to_variable = { agricultural_development_process = agricultural_development_mod }
	add_to_variable = { military_development_process = military_development_mod }
	add_to_variable = { living_standard_development_process = living_standard_development_mod }
	add_to_variable = { environment_development_process = environment_development_mod }
