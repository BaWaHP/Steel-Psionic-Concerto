ZBC_station_system_update = {
	set_variable = { ZBC_station_dynamic_modifier_ppf = 0 }
	set_variable = { ZBC_station_dynamic_modifier_cgf = 0 }
	every_owned_state = {
		if = {
			limit = { has_dynamic_modifier = { modifier = ZBC_outstation } }
			ZBC = {
				add_to_variable = { ZBC_station_dynamic_modifier_ppf = -0.01 }
				add_to_variable = { ZBC_station_dynamic_modifier_cgf = -0.01 }
			}
		}
	}
	clamp_variable = {
		var = ZBC_station_dynamic_modifier_ppf
		min = -0.15
		max = 0
	}
	clamp_variable = {
		var = ZBC_station_dynamic_modifier_cgf
		min = -0.15
		max = 0
	}
}
