NCD_skilled_worker_idea_up = {
	custom_effect_tooltip = NCD_skilled_worker_idea_up_tt
	if = {
		limit = { has_idea = NCD_skilled_worker_amount_1 }
		swap_ideas = {
			remove_idea = NCD_skilled_worker_amount_1
			add_idea = NCD_skilled_worker_amount_2
		}
	}
	else_if = {
		limit = { has_idea = NCD_skilled_worker_amount_2 }
		swap_ideas = {
			remove_idea = NCD_skilled_worker_amount_2
			add_idea = NCD_skilled_worker_amount_3

		limit = { has_idea = NCD_skilled_worker_amount_3 }

NCD_interal_conflicts_init = {
	set_variable = { NCD_interal_conflicts_var_1_1 = 15 }	#国王运动 - 政治影响力
	set_variable = { NCD_interal_conflicts_var_1_2 = 10 }	#国王运动 - 经济影响力
	set_variable = { NCD_interal_conflicts_var_2 = 10 }	#救国阵线 - 党派支持度
	set_variable = { NCD_interal_conflicts_var_3 = 20 }	#联合阵线 - 游击影响度

	activate_mission = NCD_interal_conflicts_take_actions_1
	activate_mission = NCD_interal_conflicts_take_actions_3
	activate_mission = NCD_interal_conflicts_take_actions_4

	set_variable = { NCD_demand_mil_support_count = 0 }
	set_variable = { NCD_interest_for_weapons_count = 0 }
}

NCD_interal_conflicts_var_clamp_effect = {
	clamp_variable = { var = NCD_interal_conflicts_var_1_1 min = 0 max = 35 }
	clamp_variable = { var = NCD_interal_conflicts_var_1_2 min = 0 max = 35 }
	clamp_variable = { var = NCD_interal_conflicts_var_2 min = 0 max = 50 }
	clamp_variable = { var = NCD_interal_conflicts_var_3 min = 0 max = 50 }
	if = {
		limit = { NCL = { exists = yes } }
		NCL_silt_double_beam_effect = yes
	}
}

NCL_silt_double_beam_remove = {
	NCL = {
		if = {
			limit = { has_idea = NCL_silt_double_beam_1 }
			remove_ideas = NCL_silt_double_beam_1
		}
		if = {
			limit = { has_idea = NCL_silt_double_beam_2 }
			remove_ideas = NCL_silt_double_beam_2
		}
		if = {
			limit = { has_idea = NCL_silt_double_beam_3 }
			remove_ideas = NCL_silt_double_beam_3
		}
		if = {
			limit = { has_idea = NCL_silt_double_beam_4 }
			remove_ideas = NCL_silt_double_beam_4
		}
		if = {
			limit = { has_idea = NCL_silt_double_beam_5 }
			remove_ideas = NCL_silt_double_beam_5
		}
	}
}

NCL_silt_double_beam_effect = {
	hidden_effect = {
		NCL_silt_double_beam_remove = yes
		if = {
			limit = { check_variable = { NCD_interal_conflicts_var_1_1 < 4.6 } }
			NCL = { add_ideas = NCL_silt_double_beam_1 }
		}
		else_if = {
			limit = { check_variable = { NCD_interal_conflicts_var_1_1 < 9.6 } }
			NCL = { add_ideas = NCL_silt_double_beam_2 }
		}
		else_if = {
			limit = { check_variable = { NCD_interal_conflicts_var_1_1 < 19.6 } }
			NCL = { add_ideas = NCL_silt_double_beam_3 }
		}
		else_if = {
			limit = { check_variable = { NCD_interal_conflicts_var_1_1 < 29.6 } }
			NCL = { add_ideas = NCL_silt_double_beam_4 }
		}
		else = {
			NCL = { add_ideas = NCL_silt_double_beam_5 }
		}	
	}
}
