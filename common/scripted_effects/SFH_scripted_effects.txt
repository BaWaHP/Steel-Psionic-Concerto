SFH_mermaid_influence_modifier_update = {
	clamp_variable = {
		var = SFH_mermaid_satisfaction_variable
		min = 0
		max = 100
	}
	clamp_variable = {

		min = 0
		max = 1

	#修正部分#

	if = {
		limit = {
			AND = {
				check_variable = {
					SFH_mermaid_satisfaction_variable > 79
				}
				check_variable = {
					SFH_mermaid_satisfaction_variable < 101
				}
			}
		}
		add_to_variable = {
			SFH_mermaid_influence_modifier_pp = 0.1
		}
		add_to_variable = {
			SFH_mermaid_influence_modifier_st = 0.2
		}
	}
	else_if = {
		limit = {
			AND = {
				check_variable = {
					SFH_mermaid_satisfaction_variable > 59
