		limit = { has_completed_focus = CLD_red_army_first_decade }
		CLD_red_army_var_refresh_2 = yes

		CLD_red_army_var_refresh = yes
	}

		limit = { check_variable = { current_battalions > CLD_red_army_max_battalions } }
		set_temp_variable = { battalions_dif = current_battalions }
		subtract_from_temp_variable = { battalions_dif = CLD_red_army_max_battalions }
		divide_temp_variable = { battalions_dif = 20 }
		round_variable = battalions_dif

		multiply_temp_variable = { CLD_red_army_ms_mod = -0.02 }
		add_to_variable = { CLD_red_army_ms = CLD_red_army_ms_mod }

		set_temp_variable = { CLD_red_army_aof_mod = battalions_dif }
		multiply_temp_variable = { CLD_red_army_aof_mod = -0.02 }
		add_to_variable = { CLD_red_army_aof = CLD_red_army_aof_mod }

		set_temp_variable = { CLD_red_army_ttf_mod = battalions_dif }
		multiply_temp_variable = { CLD_red_army_ttf_mod = 0.02 }
		add_to_variable = { CLD_red_army_ttf = CLD_red_army_ttf_mod }
