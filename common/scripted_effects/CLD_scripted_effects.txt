		limit = { check_variable = { CLD_new_liberated_zone_level_1 < 10 } }
		add_to_variable = { 
			var = CLD_new_liberated_zone_level_1
			value = 1 
			tooltip = CLD_new_liberated_zone_1_tt
		}
		add_to_variable = { CLD_new_liberated_zone_lm = 0.01 }
		add_to_variable = { CLD_new_liberated_zone_lrf = 0.02 }
		add_to_variable = { CLD_new_liberated_zone_lf = 0.01 }
		add_to_variable = { CLD_new_liberated_zone_lbsf = 0.01 }
		add_to_variable = { CLD_new_liberated_zone_spsbf = 0.02 }
		add_to_variable = { CLD_new_liberated_zone_spsif = 0.02 }

			swap_ideas = {
				remove_idea = CLD_new_liberated_zone_tt_0
				add_idea = CLD_new_liberated_zone_tt_1
			}
		}
		if = {
			limit = { 
				OR = { 
					has_state_category = rural

					has_state_category = large_town
				} 
			}
			CLD = {
				add_to_variable = {
					var = agricultural_development_amount
					value = 1
					tooltip = agricultural_development_amount_tt
				}
			}
		}
		else = {
			CLD = {
				add_to_variable = {
					var = industrial_development_amount
					value = 1
					tooltip = industrial_development_amount_tt
				}
				add_to_variable = {
					var = research_development_amount
					value = 1
					tooltip = research_development_amount_tt

			}
		}
	}
}
