defined_text = {
	name = GetIndDev
	text = {
		trigger = { 
			set_temp_variable = { temp_dev = 0 }
			add_to_temp_variable = { temp_dev = industrial_development_mod }
            add_to_temp_variable = { temp_dev = modifier@monthly_industrial_development_process }
		}
		localization_key = "[?temp_dev]"
	}
	text = {
		localization_key = "0"	
	}	
}
